Ubuntu编译安装MySQL5.5

MySQL选用5.5.40的版本

A . Starup

su root

apt-get install cmake libncurses5-dev bison g++
B. install mysql

wget http://www.mysql.com/get/Downloads/MySQL-5.5/mysql-5.5.40.tar.gz/from/http://cdn.mysql.com/ -O mysql-5.5.40.tar.gz

groupadd mysql
useradd -r -g mysql mysql

tar -zvxf mysql-5.5.40.tar.gz
mkdir -pv /usr/local/mysql/data
cd mysql-5.5.40

cmake . \
-DCMAKE_INSTALL_PREFIX=/usr/local/mysql \
-DINSTALL_DATADIR=/usr/local/mysql/data \
-DMYSQL_UNIX_ADDR=/usr/local/mysql/mysqld.sock \
-DDEFAULT_CHARSET=utf8 \
-DDEFAULT_COLLATION=utf8_general_ci \
-DEXTRA_CHARSETS=all \
-DENABLED_LOCAL_INFILE=1

make
make install
参数说明：
-DCMAKE_INSTALL_PREFIX=/usr/local/mysql //安装目录
-DINSTALL_DATADIR=/usr/local/mysql/data //数据库存放目录
-DDEFAULT_CHARSET=utf8 　　　　//使用utf8字符
-DDEFAULT_COLLATION=utf8_general_ci //校验字符
-DEXTRA_CHARSETS=all 　　　　//安装所有扩展字符集
-DENABLED_LOCAL_INFILE=1 　　 //允许从本地导入数据

设置目录权限:

cd /usr/local/mysql
chown -R mysql:mysql .　//把当前目录中所有文件的所有者所有者设为root，所属组为mysql

 

C. setting mysql config

配置文件

cp support-files/my-medium.cnf /etc/my.cnf　//这个配置仅适合小内存系统(32M - 64M)

 

D. 添加到系统服务

cp support-files/mysql.server /etc/init.d/mysqld

E. 修改数据库Root用户

#
cd /usr/local/mysql
./scripts/mysql_install_db —user=mysql —ldata=/usr/local/mysql/data

启动Mysql

/etc/init.d/mysqld start

/usr/local/mysql/bin/mysql -u root -p

方法1修改密码

mysql>use mysql;
mysql>UPDATE user SET Password = password ( ‘123456’ ) WHERE User = ‘root’ ;

方法2修改密码

SET PASSWORD FOR ‘root’@’localhost’ = PASSWORD(‘123456’);

刷新缓存使之生效

mysql>flush privileges;
